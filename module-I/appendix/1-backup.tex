\begin{frame}{Backup: Topology of Lie Groups}
    The Pauli formalism provides an immediate topological classification of the analyzed groups:
    \vspace{0.3cm}
    \begin{block}{$SL(2,\mathbb{C})$ and the Hyperboloid}
        Every $A \in SL(2,\mathbb{C})$ is described by $w \in \mathbb{C}$ and $a \in \mathbb{C}^3$ such that $w^2 - a^2 = 1$. This defines a \textbf{3-dimensional complex hyperboloid} $Y^{1,2}$ embedded in $\mathbb{C}^4$.
    \end{block}

    \begin{block}{$SU(2)$ and the Sphere}
        Every $U \in SU(2)$ is described by $c \in \mathbb{R}$ and $s \in \mathbb{R}^3$ such that $c^2 + s^2 = 1$. This defines a \textbf{3-dimensional sphere} $S^3$ embedded in $\mathbb{R}^4$.
    \end{block}
\end{frame}

\begin{frame}{Backup: The Group $SO(2)$}
    \begin{definition}[10.3.1]
        Setting $J_2 = \begin{pmatrix} 0 & -1 \\ 1 & 0 \end{pmatrix}$, for $\gamma, \varsigma \in \mathbb{R}$ with $\gamma^2 + \varsigma^2 = 1$, we define $R(\gamma,\varsigma) \in SO(2)$ as:
        \[ R(\gamma,\varsigma) = \gamma 1_2 + \varsigma J_2 \]
        This condition ($\gamma^2 + \varsigma^2 = 1$) topologically identifies $SO(2)$ with the circle $S^1$.
    \end{definition}

    \begin{proposition}[10.3.5]
        The Lie algebra $\mathfrak{so}(2)$ is generated by $K(x) = xJ_2$. The exponential map is:
        \[ \exp(K(x)) = \cos x 1_2 + \sin x J_2 \]
    \end{proposition}
\end{frame}

\begin{frame}{Backup: The Group $U(1)$}
    \begin{proposition}[10.4.1 \& 10.4.2]
        The Lie algebra $\mathfrak{u}(1)$ is defined as $\mathfrak{u}(1) = \{ix | x \in \mathbb{R}\}$.\\
        Its exponential map is the well-known Euler's formula:
        \[ \exp(ix) = \cos x + i \sin x \]
    \end{proposition}
    \begin{block}{Surjectivity and Invertibility}
        The map $\exp : \{ i\varphi | -\pi \leq \varphi < \pi \} \rightarrow U(1)$ is surjective.\\
        Excluding $-1$, the map is invertible in the open interval $(-\pi, \pi)$.
    \end{block}
\end{frame}
