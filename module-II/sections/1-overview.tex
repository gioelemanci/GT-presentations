\section{Overview}

\begin{frame}{Overview \& Roadmap}
    \textbf{The Goal:} To demonstrate how the classification of continuous symmetries in the Universe reduces to a rigid problem of Euclidean geometry.

    \vspace{0.5cm}
    \textbf{Roadmap:}
    \begin{itemize}
        \item \textbf{Assumed Structure:} We assume the Cartan-Weyl decomposition of a semi-simple Lie algebra.
        \item \textbf{Lemma A (Crystallographic Restriction):} We establish the severe quantization of angles between roots.
        \item \textbf{Lemma B (Simple Roots):} We identify the fundamental building blocks of the algebra.
        \item \textbf{Main Result:} The visual classification of rank-2 algebras and the construction of \textbf{Dynkin Diagrams}.
    \end{itemize}
\end{frame}
